rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Appointments collection
    match /appointments/{appointmentId} {
      // Allow anyone to create
      allow create: if request.auth != null;
      
      // Allow anyone to read (for dashboard and admin panel)
      allow read: if true;
      
      // Allow users to update/delete their own appointments
      // OR allow status updates for admin panel (without auth)
      allow update, delete: if request.auth != null && resource.data.studentUID == request.auth.uid
                            || request.resource.data.status in ['Serving', 'Completed', 'Pending'];
    }
  }
}
